tidy_study_df = study_df %>% 
  unnest (cols = file_contents) %>% 
  separate(file_names, into = c("file_names", "."), sep = 6) %>%
  select(-.) %>% 
  pivot_longer(
    week_1:week_8,
    names_to = "week",
    values_to = "value"
  ) 

ggplot(tidy_study_df, aes(x = week, y = value)) + 
  geom_line(aes(group = file_names, color = file_names), alpha=0.8)
